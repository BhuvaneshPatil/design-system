## Indeterminate

Description...

<Canvas>
  <Story name="indeterminate">
    {() => {
      const [checkedItems, setCheckedItems] = React.useState([false, true]);
      const allChecked = checkedItems.every(Boolean);
      const isIndeterminate = checkedItems.some(Boolean) && !allChecked;
      return (
        <>
          <Checkbox
            id="parent"
            name="parent"
            indeterminate={isIndeterminate}
            onChange={(e) => setCheckedItems([e.target.value, e.target.value])}
            value={allChecked}
          />
          <label htmlFor="parent">Parent</label>
          <div style={{ padding: 10 }}>
            <Checkbox
              id="child1"
              name="child1"
              onChange={(e) => setCheckedItems([e.target.value, checkedItems[1]])}
              value={checkedItems[0]}
            />
            <label htmlFor="child1">Child 1</label>
          </div>
          <div style={{ padding: 10 }}>
            <Checkbox
              id="child2"
              name="child2"
              onChange={(e) => setCheckedItems([checkedItems[0], e.target.value])}
              value={checkedItems[1]}
            />
            <label htmlFor="child2">Child 2</label>
          </div>
        </>
      );
    }}
  </Story>
</Canvas>
