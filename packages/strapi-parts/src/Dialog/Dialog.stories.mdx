    {() => {
      const [isVisible, setIsVisible] = useState(false);
      return (
        <>
          <Button onClick={() => setIsVisible(true)}>Click me</Button>
          {isVisible && (
            <Dialog
              onClose={() => setIsVisible(false)}
              title="Confirmation"
              labelledBy="confirmation"
              describedBy="confirm-description"
            >
              <DialogBody icon={<AlertWarningIcon />}>
                <Stack size={2}>
                  <Row justifyContent="center">
                    <Text id="confirm-description">Are you sure you want to delete this?</Text>
                  </Row>
                </Stack>
              </DialogBody>
              <DialogFooter
                startAction={
                  <Button onClick={() => setIsVisible(false)} variant="tertiary">
                    Cancel
                  </Button>
                }
                endAction={
                  <Button variant="danger-light" startIcon={<DeleteIcon />}>
                    Confirm
                  </Button>
                }
              />
            </Dialog>
          )}
        </>
      );
    }}
