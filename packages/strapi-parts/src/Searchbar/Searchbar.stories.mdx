# Searchbar

[View source](https://github.com/strapi/parts/tree/develop/packages/strapi-parts/src/Searchbar)

## Imports

```js
import { Searchbar } from '@strapi/parts/Searchbar';
```

## Usage


### Searchbar default

The default searchbar

<Canvas>
  <Story name="base">
    {() => {
      const [value, setValue] = useState('');
      return (
        <SearchForm>
          <Searchbar
            name="searchbar"
            onClear={() => setValue('')}
            value={value}
            onChange={(e) => setValue(e.target.value)}
            clearLabel="Clearing the plugin search"
            placeholder="e.g: strapi-plugin-abcd"
          >
            Searching for a plugin
          </Searchbar>
        </SearchForm>
      );
    }}
  </Story>
</Canvas>
