      const [isVisible, setIsVisible] = useState(true);

          {isVisible && (
            <ConfirmationDialog onClose={() => {}} title="Confirmation">
              <ConfirmationDialogBody icon={<AlertWarningIcon />}>
                <Stack size={2}>

                  </Row>
                </Stack>
              </ConfirmationDialogBody>
              <ConfirmationDialogFooter

                endAction={
                  <Button variant="danger-light" startIcon={<DeleteIcon />}>
                    Confirm
                  </Button>
                }
              />
            </ConfirmationDialog>
          )}
        </>
      );
    }}
